~~~python
class Solution:
    def canVisitAllRooms(self, rooms: List[List[int]]) -> bool:
        ret = [False] * len(rooms)
        ret[0] = True
        visited = set()
        def foo(num):
            ret[num] = True
            visited.add(num)
            for keys in rooms[num]:
                if keys not in visited:
                    foo(keys)
        foo(0)
        return True if False not in ret else False
~~~

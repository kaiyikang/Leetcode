class UnionFind:
    def __init__(self):
        self.parent = {}
        self.value = {}
    
    def find(self,x):
        root = x
        base = 1

        while self.parent[root]:
            root = self.parent[root]
            base *= self.value[root]

        while x != root:
            org_root = self.parent[x]
            
            self.value[x] *= base
            base /= self.value[org_root]

            self.parent[x] = root
            x = org_root

        return root

    def merge(self,x,y,val):
        rootx, rooty = self.find(x),self.find(y)
        if rootx != rooty:
            self.parent[rootx] = rooty
            self.value[rootx] = self.value[y] * val / self.value[x]

    def is_connected(self,x,y):
        return x in self.value and y in self.value and self.find(x) == self.find(y)
    
    def add(self,x):
        if x not in self.parent:
            self.parent[x] = None
            self.value[x] = 1.0
            

class Solution:
    def calcEquation(self, equations: List[List[str]], values: List[float], queries: List[List[str]]) -> List[float]:
        uf = UnionFind()
        for (a,b),val in zip(equations,values):
            uf.add(a)
            uf.add(b)
            uf.merge(a,b,val)

        ret = [-1.0] * len(queries)
        for i,(a,b) in enumerate(queries):
            if uf.is_connected(a,b):
                ret[i] = uf.value[a] / uf.value[b]

        return ret

class Solution:
    def findCircleNum(self, isConnected: List[List[int]]) -> int:
        
        def dfs(i):
            for j in range(n):
                if isConnected[i][j] and j not in visited:
                    visited.add(j)
                    dfs(j)

        ans = 0
        n = len(isConnected)
        visited = set()
        for i in range(n):
            if i not in visited:
                dfs(i)
                ans += 1
        return ans
